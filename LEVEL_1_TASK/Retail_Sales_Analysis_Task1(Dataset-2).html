import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# 1. Load the dataset
df = pd.read_csv('menu.csv')

# 2. Basic Data Cleaning & Inspection
print("Dataset Shape:", df.shape)
print("\nMissing Values:\n", df.isnull().sum())
# Display the first few rows to verify data
print("\nFirst 5 rows:\n", df.head())

# 3. Descriptive Statistics
# Focused on Calories, Total Fat, and Protein
stats_summary = df[['Calories', 'Total Fat', 'Protein', 'Carbohydrates', 'Sodium']].describe()
print("\nNutritional Statistical Summary:\n", stats_summary)

# 4. Data Visualization

# A. Average Calories per Category
plt.figure(figsize=(12, 6))
sns.barplot(x='Category', y='Calories', data=df, errorbar=None, palette='viridis')
plt.title('Average Calories by Menu Category')
plt.xticks(rotation=45)
plt.show()

# B. Relationship between Calories and Total Fat
plt.figure(figsize=(10, 6))
sns.scatterplot(x='Calories', y='Total Fat', hue='Category', data=df)
plt.title('Calories vs. Total Fat across Categories')
plt.show()

# C. Distribution of Sodium Levels
plt.figure(figsize=(10, 6))
sns.histplot(df['Sodium'], bins=20, kde=True, color='red')
plt.title('Distribution of Sodium Content in Menu Items')
plt.xlabel('Sodium (mg)')
plt.show()

# 5. Finding the "Heaviest" Items
top_calories = df.nlargest(5, 'Calories')[['Item', 'Category', 'Calories']]
print("\nTop 5 Highest Calorie Items:\n", top_calories)
