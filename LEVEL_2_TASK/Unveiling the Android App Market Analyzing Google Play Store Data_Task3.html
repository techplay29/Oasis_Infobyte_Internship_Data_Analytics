import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# 1.DATA PREPARATION 
# Clean and correct data types for accuracy
try:
    apps = pd.read_csv('apps.csv')
    reviews = pd.read_csv('user_reviews.csv')
except FileNotFoundError:
    apps = pd.read_csv('apps.csv.zip/apps.csv')
    reviews = pd.read_csv('user_reviews.csv.zip/user_reviews.csv')

apps['Installs'] = apps['Installs'].str.replace(',', '').str.replace('+', '').astype(int)

# Correcting Price: Removing '$' and converting to float
apps['Price'] = apps['Price'].str.replace('$', '').astype(float)

# Handling missing values
apps = apps.dropna(subset=['Rating'])
apps['Size'] = apps['Size'].fillna(apps['Size'].mean())

# 5.INTERACTIVE VISUALIZATION
# Integrating insights from "Understanding Data Visualization" course
# Focus: Data-Ink Ratio, Hierarchy, and Color Theory
plt.style.use('seaborn-v0_8')
fig, axes = plt.subplots(2, 2, figsize=(15, 12))

# Plot 1: Category Distribution (Market Dynamics)
sns.barplot(x=category_counts.values[:10], y=category_counts.index[:10], ax=axes[0, 0], palette='viridis')
axes[0, 0].set_title('Top 10 App Categories by Count', fontsize=14, fontweight='bold')
axes[0, 0].set_xlabel('Number of Apps')

# Plot 2: Rating Density (Metric Analysis)
sns.histplot(apps['Rating'], bins=20, kde=True, ax=axes[0, 1], color='skyblue')
axes[0, 1].set_title('Distribution of App Ratings', fontsize=14, fontweight='bold')

# Plot 3: App Size vs Rating (Examining quality dynamics)
sns.scatterplot(x='Size', y='Rating', data=apps, alpha=0.3, ax=axes[1, 0], color='orange')
axes[1, 0].set_title('App Size vs. Rating', fontsize=14, fontweight='bold')

# Plot 4: Sentiment Polarity (User Sentiment Assessment)
sns.boxplot(x='Type', y='Sentiment_Polarity', data=merged_df, ax=axes[1, 1], palette='RdYlGn')
axes[1, 1].set_title('Sentiment Polarity: Free vs Paid Apps', fontsize=14, fontweight='bold')

plt.tight_layout()
plt.show()
